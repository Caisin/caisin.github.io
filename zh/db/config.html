# 参数配置

  1. 实时账单开关
select t.*,t.rowid from param.bs_sys_profiles t where t.profile_code='AMS_YC_SWITCH';
insert into param.bs_sys_profiles (PROFILE_TYPE, PROFILE_CODE, PROFILE_VALUE, REGION_ID, REMARK)
values ('AMS_YC_SWITCH', 'AMS_YC_SWITCH', 'true', 'ZZZZ', '实时账单查询开关');
  1. 同步/异步缴费开关
select t.*,t.rowid from param.bs_sys_profiles t where t.profile_code like '%AM_ASYNC_PAYMENT_VALID_%';
  1. 数据源配置
select *
from CFG_DB_RELAT r, cfg_db_acct a, CFG_DB_URL u
where r.URL_NAME=u.NAME
and r.DB_ACCT_CODE = a.DB_ACCT_CODE
and r.STATE='U'
and a.STATE='U'
and u.STATE='U'
--and a.DB_ACCT_CODE in ('sfframe1', 'SFCBASE') -- 数据源名称
and r.SERVER_NAME='assetprocess'; -- 进程名
  1. 抵扣期切换
--将上一个月的MONTH_ACCT_STATUS改为2
select t.*,t.rowid from param.am_cycle t where t.cycle_id in(201909,201910,201911);
  1. 批量业务配置
select t.*,t.rowid from param.bs_static t where t.type_id='BS_BATCH_TRADE_TYPE_QRY';
select t.*,t.rowid  from param.BS_IMPORT_CONFIG t where t.trade_type_code='7788';
  1. 序列规则配置
#全局配置
sequence.class=com.ai.acctcomp.dao.seq.FuncSequence,%c1%dyyMM%s10|50
#对某个序列单独配置
sequence.class.SEQ_REQUEST_ID=com.ai.acctcomp.dao.seq.FuncSequence,%dd%s10|50
#参数说明:
#%c1:表示1位中心编号
#%dyyMM: %d时间格式 
#%s10: 10位的数据库序列 不足10的补0
#|50:一次获取缓存50个
#%c1%dyyMM%s10|50 :1位中心编码+4位年月+10位的序列|每次获取50个序列

  1. 可传账本科目业务类型配置 select * from param.BS_ENTITY_TAG_REL t where t.TAG_CODE='ASM_SPE_BUSINESS_TYPE_ID';

  2. 路由

    ipconfig /all 查看网卡的 子网掩码 和 默认网关
route add 静态路由地址 mask 子网掩码 默认网关 /p
route add 172.21.164.0 mask 255.255.255.0 192.168.137.1 /p
route add 172.25.132.0 mask 255.255.255.0 192.168.137.1 /p
route add 172.25.136.0 mask 255.255.255.0 192.168.137.1 /p
route add 172.25.137.0 mask 255.255.255.0 192.168.137.1 /p
route add 172.25.138.0 mask 255.255.255.0 192.168.137.1 /p
route delete 172.21.164.0
route delete 172.25.132.0
route delete 172.25.136.0
route delete 172.25.137.0
route delete 172.25.138.0
